let content_type_from_filename name =
    try let last_dot = String.rindex name '.' in
        match String.sub name (last_dot+1) (String.length name - last_dot - 1) with
        | "txt"   -> "text/plain"
        | "html"  -> "text/html"
        | "js"    -> "text/javascript"
        | "css"   -> "text/css"
        | "csv"   -> "text/csv"
        | "xhtml" -> "application/xhtml+xml"
        | "png"   -> "image/png"
        | "ico"   -> "image/ico"
        | "jpg"
        | "jpeg"  -> "image/jpeg"
        | _       -> "text/plain"
    with Not_found -> "text/plain"
