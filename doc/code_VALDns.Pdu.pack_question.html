        let len = String.length name in
        if len = 0 || name.[0] = '.' || name.[len-1] <> '.' then (
            err "Dns: Bad name"
        ) else (
            let str = String.create (len + 1 + 4) in
            Option.Monad.bind (pack_name name 0 str 0) (fun o ->
                pack_n16 (qtype :> int) str o ;
                pack_n16 qclass str (o + 2) ;
                Some str)
        )
